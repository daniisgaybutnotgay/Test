# Send HTTP CONNECT to a known destination (e.g., google.com:443)
connect_req = b'CONNECT www.google.com:443 HTTP/1.1\r\nHost: project-stealth.xyz\r\n\r\n'
s.send(connect_req)

response = s.recv(1024)
print(f'CONNECT response: {response[:200]}')

# If 200 OK, try TLS through the tunnel
if b'200' in response:
    import ssl
    context = ssl.create_default_context()
    context.check_hostname = False
    context.verify_mode = ssl.CERT_NONE
    ss = context.wrap_socket(s, server_hostname='www.google.com')
    ss.send(b'GET / HTTP/1.1\r\nHost: www.google.com\r\nConnection: close\r\n\r\n')
    data = ss.recv(1024)
    print(f'Tunneled response: {data[:150]}')

else:
