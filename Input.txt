echo "=== stack scan ===" && python3 -c "
import os
fd = os.open('/proc/1/mem', os.O_RDONLY)
# Scan stack region
start = 0x7ffd3c0da000
end = 0x7ffd3c14f000
os.lseek(fd, start, os.SEEK_SET)
size = end - start
data = os.read(fd, size)
for needle in [b'192.0.2.1', b'websocket', b'Upgrade', b'process_api', b'claude_chat_', b'mount_root', b'control']:
    idx = data.find(needle)
    if idx != -1:
        context = data[max(0,idx-32):idx+64]
        print(f'Found {needle} at offset {hex(start+idx)}: {context}')
os.close(fd)
print('Stack scan complete')
