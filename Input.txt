for host in google.com cloudflare.com; do
  # Test with fake SNI
  sni_code=$(curl -s -o /dev/null -w "%{http_code}" --max-time 3 -H "Host: $host" https://github.com/ 2>/dev/null)
  echo "$host (SNI via github.com): $sni_code"
done

# Test DNS over HTTPS (DoH)
doh_code=$(curl -s -o /dev/null -w "%{http_code}" --max-time 3 https://cloudflare-dns.com/dns-query 2>/dev/null)
echo "DoH (cloudflare-dns.com): $doh_code"

# Test if we can resolve arbitrary domains
nslookup google.com 2>/dev/null | head -3 || dig +short google.com 2>/dev/null | head -3 || echo "no DNS tools"
