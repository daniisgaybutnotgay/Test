import os, socket, ctypes

libc = ctypes.CDLL(None)
PID = 1

# Try to dup the WebSocket socket from PID 1
# First, check if we can use pidfd_getfd
PIDFD_GETFD = 438  # syscall number on x86_64

# Open a pidfd
pidfd = os.pidfd_open(PID, 0)
print(f'pidfd_open: {pidfd}')

# Try to duplicate fd 11 (the WebSocket socket)
try:
    dup_fd = libc.syscall(PIDFD_GETFD, pidfd, 11, 0)
    print(f'pidfd_getfd(11): {dup_fd} errno={ctypes.get_errno()}')
    if dup_fd > 0:
        s = socket.socket(fileno=dup_fd)
        print(f'Socket type: {s.getsockopt(socket.SOL_SOCKET, socket.SO_TYPE)}')
        s.close()
except Exception as e:
    print(f'pidfd_getfd failed: {e}')

os.close(pidfd)

# Also test splice/tee on the socket
try:
    # Create a pipe
    r, w = os.pipe()
    # Try to splice from PID 1's socket to our pipe
    SPLICE_F_MOVE = 1
    result = libc.splice(11, 0, w, 0, 4096, SPLICE_F_MOVE)
    print(f'splice from fd 11: {result} errno={ctypes.get_errno()}')
    os.close(r)
    os.close(w)
except Exception as e:
    print(f'splice test: {e}')
