import os, socket, ctypes, select

libc = ctypes.CDLL(None)
PID = 1

# Re-acquire the socket
pidfd = os.pidfd_open(PID, 0)
dup_fd = libc.syscall(438, pidfd, 11, 0)
os.close(pidfd)

s = socket.socket(fileno=dup_fd)
s.setblocking(False)

# Send a valid WebSocket ping frame (FIN=1, opcode=9, no mask, len=0)
# Frame: 0x89 0x00
ping_frame = bytes([0x89, 0x00])
try:
    sent = s.send(ping_frame)
    print(f'Ping sent: {sent} bytes')
except Exception as e:
    print(f'Ping send failed: {e}')

# Wait up to 2 seconds for pong or any data
ready, _, _ = select.select([s], [], [], 2)
if ready:
    try:
        data = s.recv(4096)
        print(f'Response: {len(data)} bytes: {data[:20].hex()}')
        if data[:2] == bytes([0x8A, 0x00]):
            print('SUCCESS: Received WebSocket pong — control channel is interactive!')
        else:
            print(f'Unexpected response: {data[:20]}')
    except Exception as e:
        print(f'Recv error: {e}')
else:
    print('No response within 2s (socket may be idle or gateway ignored ping)')

# Check if socket is still connected
try:
    s.getpeername()
    print('Socket still connected')
except Exception as e:
    print(f'Socket disconnected: {e}')

s.close()
print('Test complete')
