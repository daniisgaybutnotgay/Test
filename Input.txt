import os, socket, ctypes, select

libc = ctypes.CDLL(None)
PID = 1

# Re-acquire the socket
pidfd = os.pidfd_open(PID, 0)
dup_fd = libc.syscall(438, pidfd, 11, 0)
os.close(pidfd)

print(f'Stolen fd: {dup_fd}')

# Wrap it
s = socket.socket(fileno=dup_fd)

# Check socket state
print(f'Socket family: {s.family}')
print(f'Socket type: {s.type}')

# Try non-blocking recv
s.setblocking(False)
try:
    data = s.recv(4096)
    print(f'Recv data: {len(data)} bytes: {data[:100]}')
except BlockingIOError:
    print('No data available (non-blocking)')
except Exception as e:
    print(f'Recv error: {e}')

# Try to get peer address
try:
    peer = s.getpeername()
    print(f'Peer: {peer}')
except Exception as e:
    print(f'getpeername: {e}')

# Try to get local address
try:
    local = s.getsockname()
    print(f'Local: {local}')
except Exception as e:
    print(f'getsockname: {e}')

# Check socket options
try:
    sndbuf = s.getsockopt(socket.SOL_SOCKET, socket.SO_SNDBUF)
    rcvbuf = s.getsockopt(socket.SOL_SOCKET, socket.SO_RCVBUF)
    print(f'Send buffer: {sndbuf}, Recv buffer: {rcvbuf}')
except Exception as e:
    print(f'Buffer check: {e}')

s.close()
print('Socket test complete')
